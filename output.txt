select a.author, 
a.title,
 a.article_id, 
 a.topic,
  a.created_at, 
  a.votes,
   a.article_img_url,
    count(c.article_id) as comment_count
from articles a
join comments c
on a.article_id = c.article_id
group by
a.author,
a.title,
a.article_id,
a.topic,
a.created_at,
a.votes,
a.article_img_url
order by created_at desc;
